// Initialize the chat
const chat = new SpawnKitChat(containerElement, {
  relayUrl: 'http://localhost:18790',
  theme: 'dark',
  onSendMessage: (text, agentId) => console.log('Sent:', text),
  collapsed: false
});

// Control the chat
chat.toggle();                    // Show/hide
chat.setTheme('light');          // Change theme
chat.addMessage({                // Add message
  agent: 'Forge',
  text: 'Task completed!',
  type: 'agent'
});
chat.destroy();                  // Cleanup
